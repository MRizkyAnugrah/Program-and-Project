SUARA MONOKROMATIS¶

In [2]:
sampling = 8000
amplitudo = 1.0/sampling
duration = np.arange(0,1,amplitudo)
frekuensi1 = 500
frekuensi2 = 550
frekuensi3 = 600
x1 = np.cos(2*np.pi*frekuensi1*duration)
x2 = np.cos(2*np.pi*frekuensi2*duration)
x3 = np.cos(2*np.pi*frekuensi3*duration)
